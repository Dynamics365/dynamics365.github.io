This is simple example about reading excel file and save it to the table in Dynamics AX 2012.
I have one excel file like this.
This is Job that reads this excel sheet.

  1. static void ExcelJobProduct(Args _args)
  2. {
  3.     SysExcelApplication application;
  4.     SysExcelWorkbooks workbooks;
  5.     SysExcelWorkbook workbook;
  6.     SysExcelWorksheets worksheets;
  7.     SysExcelWorksheet worksheet;
  8.     SysExcelCells cells;
  9.     COMVariantType type;
  10.     ProductType prodType;
  11.     FileName filename;
  12.     Product Product;
  13.     ProductId prodId;
  14.     Name productType;
  15.     int row = 1;
  16.     str ProductName;
  17.     real Price;
  18.     str _ProductCode;
  19.     application = SysExcelApplication::construct();
  20.     workbooks = application.workbooks();
  21.  
  22.     //specify the file path that you want to read
  23.     filename = "C:Task_2_DataSet_Normalized.xlsx";
  24.     try
  25.     {
  26.         //  Adds the file as the first document in the collection.
  27.         workbooks.open(filename);
  28.     }
  29.     catch (Exception::Error)
  30.     {
  31.         throw error("File cannot be opened.");
  32.     }
  33.     workbook = workbooks.item(1);
  34.     worksheets = workbook.worksheets();
  35.     worksheet = worksheets.itemFromNum(5);   // represents the current worksheet in my case it's (5)
  36.     cells = worksheet.cells();
  37.     // Fetches data from each cell that contains data.
  38.     do
  39.     {
  40.         row++;
  41.  
  42.         Price = cells.item(row, 4).value().double();
  43.         prodId = int642str(cells.item(row, 1).value().double());
  44.         ProductName = cells.item(row, 2).value().bstr();
  45.         productType = cells.item(row, 3).value().bStr();
  46.         // Insert data into Product table based on the ProductType from ProductType table.
  47.         select prodType
  48.          where prodType.ProductType == productType;
  49.         if (prodType.RecId != 0)
  50.         {
  51.             ttsBegin;
  52.             Product.ProductCode = prodId;
  53.             Product.Name = ProductName;
  54.             Product.ProductType = prodType.RecId;
  55.             Product.Price = Price;
  56.             Product.insert();
  57.             ttsCommit;
  58.         }
  59.         type = cells.item(row + 1, 1).value().variantType();
  60.     }
  61.  
  62.     // Runs until the COMVarientType doesnot contains a data field.
  63.     while (type != COMVariantType::VT_EMPTY);
  64.  
  65.     // Closes the Instance of Excel.
  66.     application.quit();
  67. }

Here're step:
  1. SysExcelApplication  represents an instance of Excel .
  2. SysExcelWorkbooks  provides collection of Excel documents stored in this Class.
  3. SysExcelWorkbook    provides a reference to the opened document.
  4. SysExcelWorksheets represents a collection of all the worksheets in a document.
  5. SysExcelWorksheet   provides a reference to a single worksheet.
  6. SysExcelCells             provides reference to the collection of cells within a sheet.
  7. COMVariantType        COMVarient Class is used to store various types of data .
