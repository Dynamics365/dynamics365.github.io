  32.     do
  33.     {
  34.         row++;
  35.         itemId = cells.item(row, 1).value().bStr();

  1. static void Write2ExcelFile(Args _args)
  2. {
  3.     InventTable inventTable;
  4.     SysExcelApplication application;
  5.     SysExcelWorkbooks workbooks;
  6.     SysExcelWorkbook workbook;
  7.     SysExcelWorksheets worksheets;
  8.     SysExcelWorksheet worksheet;
  9.     SysExcelCells cells;
  10.     SysExcelCell cell;
  11.     int row;
  12.     ;
  13.     application = SysExcelApplication::construct();
  14.     workbooks = application.workbooks();
  15.     workbook = workbooks.add();
  16.     worksheets = workbook.worksheets();
  17.     worksheet = worksheets.itemFromNum(1);
  18.     cells = worksheet.cells();
  19.     cells.range('A:A').numberFormat('@');
  20.  
  21.     cell = cells.item(1, 1);
  22.     cell.value("Item");
  23.     cell = cells.item(1, 2);
  24.     cell.value("Name");
  25.     row = 1;
  26.     while select inventTable
  27. {
  28.         row++;
  29.         cell = cells.item(row, 1);
  30.         cell.value(inventTable.ItemId);
  31.         cell = cells.item(row, 2);
  32.         cell.value(inventTable.ItemName);
  33.     }
  34.     application.visible(true);
  35. }

How it works 

1. Use SysExcelApplication class to create excel file.
2. Use SysExcelWorkbooks and SysExcelWorkbook to create a blank workbook(by default 3 worksheets will be available).
3. Use SysExcelWorkSheets to select worksheet for writing data.
4. SysExcelCells to select the cells in the excel for writing the data.
5. SysExcelCell to write the data in the selected cells. 
6. Once you done with write operation use SysExcelApplication.visible to open file.
